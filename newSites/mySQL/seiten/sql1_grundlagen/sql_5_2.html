UNION auf mehrere Spalten:

 SELECT
            artikelbezeichnung,
            kategorie,
            legierung
         FROM produkt
         UNION
         SELECT
            artikelbezeichnung,
            kategorie,
            legierung
         FROM produktarchiv 
         
         SELECT * FROM produkt;
