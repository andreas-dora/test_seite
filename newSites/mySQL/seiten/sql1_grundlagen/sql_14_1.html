    artikelbezeichnung
    LOWER(artikelbezeichnung) AS artikelbezeichnungklein

    produkt;

SUBSTRING

SELECT
    artikelbezeichnung
    SUBSTRING(artikelbezeichnung FROM 1 FOR 8) AS extrakt
FROM
    produkt;

CONCAT

    artikelbezeichnung,
    kategorie,
    legierung,
    CONCAT(artikelbezeichnung,' ',kategorie,' ',legierung)
        AS produktinfo
FROM
    produkt;
